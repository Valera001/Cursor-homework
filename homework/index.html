Перелік основних команд:

git config --global user.name blabla - задати им'я користувача в системі (тільки один раз);

git config --global user.email blabal@bla - задати email користувача в системі (тільки один раз);

git init - ініціалізуе новий (пустий) репозиторій git;

git status - перевірка стану файлів репозиторія (які файли змінені, але не додані до індексу);

git add - додае до сисеми контролю версій (в розділ проіндексованих файлів) змінені файли;

git commit -запис індексованих змін до репозиторію;

git log - перегляд історії комітів;

git checkout - переключення між гілками;

git branch - створюе нову гілку (git branch [branch-name]);

git push -u - вивантаження коміти у віддалений репозиторій;

git pull - завантажує всі зміни з віддаленого репозиторію та об'єднує їх з локальним;

git merge - злиття однієї або кількох гілок у поточну;

git clone - створення копії віддаленого репозиторія.

.gitignore файли

.gitignore файл - це файли які не відслідковуються git та не попадають в коміти;

Advanced

git revert -відміняє останній коміт, створюючи новий який включає всі відмінені зміни;

git reset - скасування локальних змін у репозиторії git;

git rebase - зміни, що були в комітах одної гілки відтворити на іншій гілці;

git cherry-pick - вибиратє довільні коміти Git за посиланням та додаватє їх;

git stash - тимчасово архівує зміни.
